    【Eigen】Cholesky分解的实现

    #include <iostream>
#include <Eigen/Sparse>
#include <Eigen/Core>
#include <random>
#include <vector>
#include <cmath>

// Cholesky分解函数，输入A为对称正定稀疏矩阵，返回下三角L（稀疏）
Eigen::SparseMatrix<double> CholeskyDecompose(const Eigen::SparseMatrix<double>& A) {
    int n = A.rows();
    Eigen::SparseMatrix<double> L(n, n);
    // 计算L11
    L.coeffRef(0, 0) = sqrt(A.coeff(0, 0));
    //计算第一列剩下的元素
    for (int i=1;i<n;i++){
        L.coeffRef(i,0)=A.coeff(i,0)/L.coeff(0,0);
    }
    //逐列计算剩下的元素
    for(int i=2;i<=n;i++){
        //计算L_ii
        float temp=0.0;
        for (int k=1;k<=i-1;k++){
            temp+=L.coeff(i-1,k-1)*L.coeff(i-1,k-1);
        }
        L.coeffRef(i-1,i-1)=sqrt(A.coeff(i-1,i-1)-temp);

        //计算第i列第j行元素
        for (int j=i+1;j<=n;j++){
            float temp2=0.0;
            for (int k=1;k<=i-1;k++){
                temp2+=L.coeff(j-1,k-1)*L.coeff(i-1,k-1);
            }
            L.coeffRef(j-1,i-1)=(A.coeff(j-1,i-1)-temp2)/L.coeff(i-1,i-1);
        }
    }
    return L;
}

int main() {
    int n = 5;
    float maxVal = 500;
    // 生成一个随机的下三角稀疏矩阵L
    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> dis(0, maxVal);
    Eigen::SparseMatrix<double> L_true(n, n);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j <= i; ++j) {
            double val = dis(gen);
            L_true.coeffRef(i, j) = val; // 只在下三角部分赋值
        }
    }

    // 计算A = LL^T
    Eigen::SparseMatrix<double> A = L_true * L_true.transpose();

    std::cout << "----生成的L-----:\n";
    for (int i = 0; i < L_true.rows(); i++) {
        for (int j = 0; j < L_true.cols(); j++) {
            std::cout <<L_true.coeff(i, j) <<",";
        }
        std::cout << std::endl;
    }

    std::cout << "\n----LL^T组合成的A-----:\n";
    for (int i = 0; i < A.rows(); i++) {
        for (int j = 0; j < A.cols(); j++) {
            std::cout <<A.coeff(i, j) <<",";
        }
        std::cout << std::endl;
    }

    Eigen::SparseMatrix<double> L = CholeskyDecompose(A);

    std::cout << "\n----从A分解得到的L-------:\n";
    for (int i = 0; i < L.rows(); i++) {
        for (int j = 0; j < L.cols(); j++) {
            std::cout <<L.coeff(i, j) <<",";
        }
        std::cout << std::endl;
    }
    return 0;
}
